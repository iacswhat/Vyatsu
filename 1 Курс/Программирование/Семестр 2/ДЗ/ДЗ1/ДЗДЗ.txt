var
  f1, f2: text;
  s: string;
  i, k: integer;
  e: char;

begin
  e := '2';
  assign(f1, 'vvod.txt');
  assign(f2, 'vivod.txt');
  reset(f1);
  rewrite(f2);
  while not eof(f1) do 
  begin
    k := 0;
    readln(f1, s);
    for i := 1 to length(s) do
    begin
      if s[i] = e then
        k := k + 1;
    end;
    if k = 0 then
    begin
      writeln(f2, s);
    end;
  end;
  close(f1);
  close(f2);
end.
